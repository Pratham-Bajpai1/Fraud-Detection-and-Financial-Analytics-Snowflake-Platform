USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;




CREATE OR REPLACE TABLE FACT_TRANSACTIONS AS

SELECT
    -- Transaction Identifier
    ST.TRANSACTION_HASH,

    -- Dimension Keys
    DC.CUSTOMER_KEY,
    DM.MERCHANT_KEY,
    DT.TIME_KEY,
    DL.LOCATION_KEY,
    DTT.TRANSACTION_TYPE_KEY,

    -- Transaction Measures
    ST.TRANSACTION_AMOUNT,

    ST.CUSTOMER_OLD_BALANCE,
    ST.CUSTOMER_NEW_BALANCE,

    ST.MERCHANT_OLD_BALANCE,
    ST.MERCHANT_NEW_BALANCE,

    ST.CUSTOMER_BALANCE_CHANGE,

    -- Transaction Attributes
    ST.HIGH_VALUE_FLAG,
    ST.RISK_CATEGORY,

    ST.IS_FRAUD,
    ST.IS_FLAGGED_FRAUD,

    ST.TRANSACTION_TIMESTAMP,

    CURRENT_TIMESTAMP() AS CREATED_AT

FROM SILVER_LAYER.SILVER_TRANSACTIONS ST

INNER JOIN DIM_CUSTOMER DC
    ON ST.CUSTOMER_ID = DC.CUSTOMER_ID

INNER JOIN DIM_MERCHANT DM
    ON ST.MERCHANT_ID = DM.MERCHANT_ID

INNER JOIN DIM_TIME DT
    ON TO_NUMBER(
        TO_CHAR(ST.TRANSACTION_TIMESTAMP, 'YYYYMMDDHH24')
    ) = DT.TIME_KEY

INNER JOIN DIM_LOCATION DL
    ON ST.CUSTOMER_ID = DL.CUSTOMER_ID

INNER JOIN DIM_TRANSACTION_TYPE DTT
    ON ST.TRANSACTION_TYPE = DTT.TRANSACTION_TYPE;




-- Validation Queries

SELECT COUNT(*)
FROM FACT_TRANSACTIONS;  // 6362620


-- Validate Transaction Type Join
SELECT
    TRANSACTION_TYPE_KEY,
    COUNT(*) AS TOTAL_TRANSACTIONS
FROM FACT_TRANSACTIONS
GROUP BY TRANSACTION_TYPE_KEY
ORDER BY TRANSACTION_TYPE_KEY;


-- Null Key Validation
SELECT
    COUNT_IF(CUSTOMER_KEY IS NULL) AS NULL_CUSTOMERS,
    COUNT_IF(MERCHANT_KEY IS NULL) AS NULL_MERCHANTS,
    COUNT_IF(TIME_KEY IS NULL) AS NULL_TIMES,
    COUNT_IF(LOCATION_KEY IS NULL) AS NULL_LOCATIONS,
    COUNT_IF(TRANSACTION_TYPE_KEY IS NULL) AS NULL_TRANSACTION_TYPES
FROM FACT_TRANSACTIONS;
