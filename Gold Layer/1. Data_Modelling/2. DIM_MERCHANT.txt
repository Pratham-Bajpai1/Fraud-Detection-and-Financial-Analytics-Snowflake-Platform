USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;


/*
i model merchant/receiver entities. This table enables:
merchant analytics
fraud concentration analysis
transaction destination insights
risky merchant identification
*/


CREATE OR REPLACE TABLE DIM_MERCHANT AS

WITH MERCHANT_AGGREGATES AS
(
    SELECT
        MERCHANT_ID,
        AVG(TRANSACTION_AMOUNT) AS AVG_TRANSACTION_AMOUNT,
        SUM(IS_FRAUD) AS FRAUD_TRANSACTION_COUNT,
        COUNT(*) AS TOTAL_TRANSACTIONS
    FROM SILVER_LAYER.SILVER_TRANSACTIONS
    GROUP BY MERCHANT_ID
)

SELECT
    ROW_NUMBER() OVER (ORDER BY MERCHANT_ID) AS MERCHANT_KEY,

    MERCHANT_ID,

    TOTAL_TRANSACTIONS,

    ROUND(AVG_TRANSACTION_AMOUNT, 2) AS AVG_TRANSACTION_AMOUNT,

    CASE
        WHEN FRAUD_TRANSACTION_COUNT >= 2
            THEN 'HIGH_RISK'
        WHEN FRAUD_TRANSACTION_COUNT > 0
            THEN 'MEDIUM_RISK'
        ELSE 'LOW_RISK'
    END AS MERCHANT_RISK_LEVEL,

    CURRENT_TIMESTAMP() AS CREATED_AT

FROM MERCHANT_AGGREGATES;




-- validation queries

-- merchant count
SELECT COUNT(*)
FROM DIM_MERCHANT;  // 2722362

-- merchant risk distribution
SELECT
    MERCHANT_RISK_LEVEL,
    COUNT(*) AS TOTAL_MERCHANTS
FROM DIM_MERCHANT
GROUP BY MERCHANT_RISK_LEVEL;

-- top fraud merchants
SELECT *
FROM DIM_MERCHANT
WHERE MERCHANT_RISK_LEVEL = 'HIGH_RISK'
LIMIT 10;
