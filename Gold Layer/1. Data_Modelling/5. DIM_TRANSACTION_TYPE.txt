USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;




CREATE OR REPLACE TABLE DIM_TRANSACTION_TYPE AS

WITH UNIQUE_TRANSACTION_TYPES AS
(
    SELECT DISTINCT
        TRANSACTION_TYPE
    FROM SILVER_LAYER.SILVER_TRANSACTIONS
)

SELECT

    ROW_NUMBER() OVER (ORDER BY TRANSACTION_TYPE) AS TRANSACTION_TYPE_KEY,

    TRANSACTION_TYPE,

    CASE
        WHEN TRANSACTION_TYPE IN ('TRANSFER', 'CASH_OUT')
            THEN 'HIGH_FRAUD_RISK'
        WHEN TRANSACTION_TYPE = 'PAYMENT'
            THEN 'LOW_FRAUD_RISK'
        ELSE 'MEDIUM_FRAUD_RISK'
    END AS TRANSACTION_RISK_CATEGORY,

    CASE
        WHEN TRANSACTION_TYPE IN ('CASH_IN', 'CASH_OUT')
            THEN 'CASH_TRANSACTION'
        ELSE 'DIGITAL_TRANSACTION'
    END AS TRANSACTION_CHANNEL,

    CURRENT_TIMESTAMP() AS CREATED_AT

FROM UNIQUE_TRANSACTION_TYPES;




-- validation queries

-- total transaction types
SELECT COUNT(*)   
FROM DIM_TRANSACTION_TYPE;   // 5


-- risk category distribution
SELECT
    TRANSACTION_RISK_CATEGORY,
    COUNT(*) AS TOTAL_TYPES
FROM DIM_TRANSACTION_TYPE
GROUP BY TRANSACTION_RISK_CATEGORY;
