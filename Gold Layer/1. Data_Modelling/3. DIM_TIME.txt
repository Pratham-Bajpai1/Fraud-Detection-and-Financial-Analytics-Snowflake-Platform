
USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;


/*
it enables:
daily analytics
monthly trends
hourly fraud analysis
dashboard filtering
KPI aggregation
*/



CREATE OR REPLACE TABLE DIM_TIME AS

SELECT DISTINCT

    TO_NUMBER(TO_CHAR(TRANSACTION_TIMESTAMP, 'YYYYMMDDHH24'))
        AS TIME_KEY,

    CAST(TRANSACTION_TIMESTAMP AS DATE)
        AS FULL_DATE,

    TRANSACTION_TIMESTAMP,

    YEAR(TRANSACTION_TIMESTAMP)
        AS YEAR,

    QUARTER(TRANSACTION_TIMESTAMP)
        AS QUARTER,

    MONTH(TRANSACTION_TIMESTAMP)
        AS MONTH,

    MONTHNAME(TRANSACTION_TIMESTAMP)
        AS MONTH_NAME,

    WEEKOFYEAR(TRANSACTION_TIMESTAMP)
        AS WEEK_NUMBER,

    DAY(TRANSACTION_TIMESTAMP)
        AS DAY_OF_MONTH,

    DAYNAME(TRANSACTION_TIMESTAMP)
        AS DAY_NAME,

    DAYOFWEEK(TRANSACTION_TIMESTAMP)
        AS DAY_OF_WEEK,

    HOUR(TRANSACTION_TIMESTAMP)
        AS HOUR_OF_DAY,

    CASE
        WHEN DAYOFWEEK(TRANSACTION_TIMESTAMP) IN (1,7)
            THEN 'WEEKEND'
        ELSE 'WEEKDAY'
    END AS DAY_TYPE

FROM SILVER_LAYER.SILVER_TRANSACTIONS;



-- Validation Queries

-- total time records, around 744 rows because: 31 days × 24 hours
SELECT COUNT(*) // 743
FROM DIM_TIME;


-- date range validation
SELECT
    MIN(TRANSACTION_TIMESTAMP),
    MAX(TRANSACTION_TIMESTAMP)
FROM DIM_TIME;

-- weekend vs weekday
SELECT
    DAY_TYPE,
    COUNT(*) AS TOTAL_HOURS
FROM DIM_TIME
GROUP BY DAY_TYPE;


-- hour distribution
SELECT
    HOUR_OF_DAY,
    COUNT(*) AS TOTAL_RECORDS
FROM DIM_TIME
GROUP BY HOUR_OF_DAY
ORDER BY HOUR_OF_DAY;
