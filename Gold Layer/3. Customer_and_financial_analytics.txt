USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;



/*
Step 1 — Top Spending Customers
This provided:
VIP customer analysis
customer profitability
fraud exposure analysis
spending behavior insights
*/


SELECT
    DC.CUSTOMER_ID,
    DC.CUSTOMER_SEGMENT,
    DL.CITY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_SPENDING,
    ROUND(
        AVG(FT.TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_AMOUNT,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

INNER JOIN DIM_LOCATION DL
    ON FT.LOCATION_KEY = DL.LOCATION_KEY

GROUP BY DC.CUSTOMER_ID, DC.CUSTOMER_SEGMENT, DL.CITY
ORDER BY TOTAL_SPENDING DESC
LIMIT 50;



/*
Step 2 — Customer Segmentation Analytics
*/

SELECT
    DC.CUSTOMER_SEGMENT,
    COUNT(DISTINCT DC.CUSTOMER_ID)
        AS TOTAL_CUSTOMERS,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    ROUND(
        AVG(FT.TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_VALUE,
    SUM(FT.IS_FRAUD)
        AS FRAUD_TRANSACTIONS

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

GROUP BY DC.CUSTOMER_SEGMENT
ORDER BY TOTAL_REVENUE DESC;




/*
Step 3 — Daily Revenue Trends
*/

SELECT
    DT.FULL_DATE,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    ROUND(
        AVG(FT.TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_VALUE,
    SUM(FT.IS_FRAUD)
        AS FRAUD_TRANSACTIONS

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY
    
GROUP BY DT.FULL_DATE
ORDER BY DT.FULL_DATE;



/*
Step 4 — Monthly Financial KPIs
*/

SELECT
    DT.YEAR,
    DT.MONTH,
    DT.MONTH_NAME,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    ROUND(
        AVG(FT.TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_VALUE,
    SUM(FT.IS_FRAUD)
        AS FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY

GROUP BY DT.YEAR, DT.MONTH, DT.MONTH_NAME
ORDER BY DT.YEAR, DT.MONTH;



/*
Step 5 — Transaction Volume by Hour
This provides:
peak transaction monitoring
fraud timing analysis
operational insights
activity heatmaps
*/


SELECT
    DT.HOUR_OF_DAY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    SUM(FT.IS_FRAUD)
        AS FRAUD_TRANSACTIONS
FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY

GROUP BY DT.HOUR_OF_DAY
ORDER BY DT.HOUR_OF_DAY;
