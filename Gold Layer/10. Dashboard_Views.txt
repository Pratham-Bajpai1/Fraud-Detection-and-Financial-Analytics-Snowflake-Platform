USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;


-- Step 1 — Executive KPI View

CREATE OR REPLACE VIEW VW_EXECUTIVE_KPIS AS

SELECT
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    ROUND(
        AVG(TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_VALUE,
    SUM(IS_FRAUD) AS TOTAL_FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT,
    COUNT_IF(HIGH_VALUE_FLAG = 1)
        AS HIGH_VALUE_TRANSACTIONS

FROM FACT_TRANSACTIONS;


-- validation
SELECT *
FROM VW_EXECUTIVE_KPIS;





-- Step 2 - FRAUD TREND ANALYTICS VIEW

CREATE OR REPLACE VIEW VW_FRAUD_TRENDS AS

SELECT
    DT.FULL_DATE,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY

GROUP BY DT.FULL_DATE
ORDER BY DT.FULL_DATE;


-- validation
SELECT * FROM VW_FRAUD_TRENDS;




-- Step 3 - FRAUD BY TRANSACTION TYPE

CREATE OR REPLACE VIEW VW_FRAUD_BY_TRANSACTION_TYPE AS

SELECT
    DTT.TRANSACTION_TYPE,
    DTT.TRANSACTION_RISK_CATEGORY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_TRANSACTION_AMOUNT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TRANSACTION_TYPE DTT
    ON FT.TRANSACTION_TYPE_KEY = DTT.TRANSACTION_TYPE_KEY

GROUP BY DTT.TRANSACTION_TYPE, DTT.TRANSACTION_RISK_CATEGORY
ORDER BY FRAUD_RATE_PERCENT DESC;


-- validation
SELECT * FROM VW_FRAUD_BY_TRANSACTION_TYPE;






-- Step 4 — Regional Fraud Analytics 

CREATE OR REPLACE VIEW VW_REGIONAL_FRAUD_ANALYTICS AS

SELECT
    DL.STATE,
    DL.CITY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_TRANSACTION_AMOUNT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_LOCATION DL
    ON FT.LOCATION_KEY = DL.LOCATION_KEY

GROUP BY DL.STATE, DL.CITY
ORDER BY FRAUD_RATE_PERCENT DESC;


-- validation
SELECT * FROM VW_REGIONAL_FRAUD_ANALYTICS;



-- Step 5 - CUSTOMER RISK PROFILE VIEW

CREATE OR REPLACE VIEW VW_CUSTOMER_RISK_PROFILES AS

SELECT
    DCRP.RISK_SEGMENT,
    COUNT(*) AS TOTAL_CUSTOMERS,
    ROUND(
        AVG(DCRP.FRAUD_RISK_SCORE),
        2
    ) AS AVG_RISK_SCORE,
    MAX(DCRP.FRAUD_RISK_SCORE) AS MAX_RISK_SCORE

FROM DIM_CUSTOMER_RISK_PROFILES DCRP

GROUP BY DCRP.RISK_SEGMENT
ORDER BY AVG_RISK_SCORE DESC;


-- validation
SELECT * FROM VW_CUSTOMER_RISK_PROFILES;






-- Step 6 - HOURLY TRANSACTION ANALYTICS

CREATE OR REPLACE VIEW VW_HOURLY_TRANSACTION_ANALYTICS AS

SELECT
    DT.HOUR_OF_DAY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_REVENUE,
    SUM(FT.IS_FRAUD)
        AS FRAUD_TRANSACTIONS

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY

GROUP BY DT.HOUR_OF_DAY
ORDER BY DT.HOUR_OF_DAY;


-- validation
SELECT * FROM VW_HOURLY_TRANSACTION_ANALYTICS;
