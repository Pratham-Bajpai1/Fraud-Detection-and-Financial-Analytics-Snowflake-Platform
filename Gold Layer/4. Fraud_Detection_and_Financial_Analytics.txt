USE DATABASE Synthetic_Financial_DB;

USE SCHEMA GOLD_LAYER;

USE WAREHOUSE Synthetic_Financial_WH;

/*
step 1 — High-Value Fraud Detection
this is one of the most common real-world fraud detection rules.
goal:
identify large suspicious transactions
analyze high-risk transaction behavior
dashboard-ready fraud insights
*/



SELECT
    FT.TRANSACTION_HASH,
    DC.CUSTOMER_ID,
    DM.MERCHANT_ID,
    DTT.TRANSACTION_TYPE,
    DL.CITY,
    DL.STATE,
    FT.TRANSACTION_AMOUNT,
    FT.RISK_CATEGORY,
    FT.IS_FRAUD,
    FT.TRANSACTION_TIMESTAMP

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

INNER JOIN DIM_MERCHANT DM
    ON FT.MERCHANT_KEY = DM.MERCHANT_KEY

INNER JOIN DIM_LOCATION DL
    ON FT.LOCATION_KEY = DL.LOCATION_KEY

INNER JOIN DIM_TRANSACTION_TYPE DTT
    ON FT.TRANSACTION_TYPE_KEY = DTT.TRANSACTION_TYPE_KEY

WHERE
    FT.HIGH_VALUE_FLAG = 1
    AND FT.IS_FRAUD = 1
ORDER BY FT.TRANSACTION_AMOUNT DESC
LIMIT 5;




-- validation query 

-- Count High Value Fraud Transactions -- 5471
SELECT COUNT(*) AS HIGH_VALUE_FRAUD_TRANSACTIONS
FROM FACT_TRANSACTIONS
WHERE HIGH_VALUE_FLAG = 1
AND IS_FRAUD = 1;


-- Fraud Percentage Among High Value Transactions This gives: fraud concentration, anomaly ratio, risk exposure metric
SELECT
    ROUND(
        100.0 *
        SUM(CASE WHEN IS_FRAUD = 1 THEN 1 ELSE 0 END)
        / COUNT(*),
        4
    ) AS FRAUD_PERCENTAGE
FROM FACT_TRANSACTIONS
WHERE HIGH_VALUE_FLAG = 1;






/*
step 2 — Fraud by Region Analysis
This powers:
fraud heatmaps
geo dashboards
regional risk analysis
anomaly monitoring
*/


SELECT
    DL.STATE,
    DL.CITY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_LOCATION DL
    ON FT.LOCATION_KEY = DL.LOCATION_KEY

GROUP BY DL.STATE, DL.CITY
ORDER BY FRAUD_RATE_PERCENT DESC, FRAUD_TRANSACTIONS DESC;





/*
step 3 — Fraud by Transaction Type
This query will strongly validate:
TRANSFER
CASH_OUT
as the riskiest transaction types.
*/


SELECT
    DTT.TRANSACTION_TYPE,
    DTT.TRANSACTION_RISK_CATEGORY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        100.0 * SUM(FT.IS_FRAUD) / COUNT(*),
        4
    ) AS FRAUD_RATE_PERCENT
FROM FACT_TRANSACTIONS FT
INNER JOIN DIM_TRANSACTION_TYPE DTT
    ON FT.TRANSACTION_TYPE_KEY = DTT.TRANSACTION_TYPE_KEY
GROUP BY
    DTT.TRANSACTION_TYPE,
    DTT.TRANSACTION_RISK_CATEGORY
ORDER BY FRAUD_RATE_PERCENT DESC;







-- Behavioral & Anomaly Detection Analytics
/* i have implemented:

/*
1. RAPID TRANSACTION VELOCITY DETECTION
This identifies:
unusually rapid transaction bursts
possible bot activity
suspicious automated transfers
mule account behavior
*/

SELECT
    DC.CUSTOMER_ID,
    DT.FULL_DATE,
    DT.HOUR_OF_DAY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.TRANSACTION_AMOUNT) AS TOTAL_TRANSACTION_AMOUNT,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

INNER JOIN DIM_TIME DT
    ON FT.TIME_KEY = DT.TIME_KEY

GROUP BY DC.CUSTOMER_ID, DT.FULL_DATE, DT.HOUR_OF_DAY
HAVING COUNT(*) >= 2
ORDER BY TOTAL_TRANSACTIONS DESC, TOTAL_TRANSACTION_AMOUNT DESC;







/* 2. SUSPICIOUS BALANCE CHANGE DETECTION
This detects:
account draining
suspicious withdrawals
potential takeover events
*/


SELECT
    FT.TRANSACTION_HASH,
    DC.CUSTOMER_ID,
    DTT.TRANSACTION_TYPE,
    FT.TRANSACTION_AMOUNT,
    FT.CUSTOMER_OLD_BALANCE,
    FT.CUSTOMER_NEW_BALANCE,
    FT.CUSTOMER_BALANCE_CHANGE,
    FT.IS_FRAUD,
    FT.TRANSACTION_TIMESTAMP

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

INNER JOIN DIM_TRANSACTION_TYPE DTT
    ON FT.TRANSACTION_TYPE_KEY = DTT.TRANSACTION_TYPE_KEY

WHERE FT.CUSTOMER_OLD_BALANCE > 0 AND FT.CUSTOMER_NEW_BALANCE = 0 AND FT.TRANSACTION_AMOUNT > 200000 AND FT.IS_FRAUD = 1
ORDER BY FT.TRANSACTION_AMOUNT DESC
LIMIT 100;
    




-- 3. TOP FRAUD-ASSOCIATED CUSTOMERS

SELECT
    DC.CUSTOMER_ID,
    DC.CUSTOMER_SEGMENT,
    DL.CITY,
    COUNT(*) AS TOTAL_TRANSACTIONS,
    SUM(FT.IS_FRAUD) AS FRAUD_TRANSACTIONS,
    ROUND(
        SUM(FT.TRANSACTION_AMOUNT),
        2
    ) AS TOTAL_TRANSACTION_AMOUNT,
    ROUND(
        AVG(FT.TRANSACTION_AMOUNT),
        2
    ) AS AVG_TRANSACTION_AMOUNT

FROM FACT_TRANSACTIONS FT

INNER JOIN DIM_CUSTOMER DC
    ON FT.CUSTOMER_KEY = DC.CUSTOMER_KEY

INNER JOIN DIM_LOCATION DL
    ON FT.LOCATION_KEY = DL.LOCATION_KEY

GROUP BY DC.CUSTOMER_ID, DC.CUSTOMER_SEGMENT, DL.CITY
HAVING SUM(FT.IS_FRAUD) > 0
ORDER BY FRAUD_TRANSACTIONS DESC, TOTAL_TRANSACTION_AMOUNT DESC;
